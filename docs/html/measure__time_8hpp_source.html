71// Measure execution time of a function
72template <typename Unit = std::chrono::milliseconds, typename Func, typename... Args>
73auto measureTime(Func&& func, Args&&... args)
74{
75 auto start = std::chrono::high_resolution_clock::now();
76 std::invoke(std::forward<Func>(func), std::forward<Args>(args)...);
77 auto end = std::chrono::high_resolution_clock::now();
78
79 auto duration = std::chrono::duration_cast<Unit>(end - start);
80 return duration;
81}
